grid_dimension = len(grid)
In [4]:
def product_horizontal(i, j):
    return reduce(mul, grid[i][j: j+4], 1)
In [5]:
def product_vertical(i, j):
    return reduce(mul, [grid[ii][j] for ii in range(i, i+4)], 1)
In [6]:
def product_diagonal(i, j):
    return reduce(mul, [grid[i+b][j+b] for b in range(4)], 1)
In [7]:
def product_antidiagonal(i, j):
    return reduce(mul, [grid[i-b][j+b] for b in range(4)], 1)

def max_of_a_cell(i, j):
    products = [(0, -1, -1, None)]
    if j < grid_dimension - 4:
        products.append((product_horizontal(i, j), i, j, 'h'))
        if i > 3:
            products.append((product_antidiagonal(i, j), i, j, 'ad'))
        if i < grid_dimension - 4:
            products.append((product_diagonal(i, j), i, j, 'd'))
    if i < grid_dimension - 4:
        products.append((product_vertical(i, j), i, j, 'v'))
    return max(products)
In [9]:
def solve():
    return max((max_of_a_cell(i, j)
               for j in range(grid_dimension)
                   for i in range(grid_dimension)))
